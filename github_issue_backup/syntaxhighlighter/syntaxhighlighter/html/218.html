Add annotation and line specific highlight feature

m42e commented on 2013-05-27 05:38:49

This patch allows to specify a highlight class for each line as you can see it here:

    <pre class="brush: c;
      annotations:{
        &quot;3&quot;:[
          {&quot;classes&quot;: [&quot;plugin_Test&quot;, &quot;type_Comment&quot;], &quot;message&quot;:&quot;Filename does not match;},
          {&quot;classes&quot;:[&quot;plugin_Test&quot;, &quot;type_Comment&quot;],  &quot;message&quot;:&quot;I'm not sure the file is properly named&quot;}
        ], 
        &quot;6&quot;:[
          {&quot;classes&quot;: [&quot;plugin_Parser&quot;, &quot;type_Exception&quot;], &quot;message&quot;:&quot;this file does not exist&quot;}
        ]
      };
      highlight:{
        &quot;10&quot;: &quot;fullycovered&quot;,
        &quot;11&quot;: &quot;partlycovered&quot;,
        &quot;12&quot;: &quot;notcovered&quot;
      };
    ">
    /*----------------------------------------------------------------------------*/
    /**
    *  \file      mynoop.c
    *
    */
    #include "something.h"
    int j;
    for(j = 0; j &lt; 1000; j++){
      asm("nop");
    }
    /*--- End of file ------------------------------------------------------------*/
    </pre>
